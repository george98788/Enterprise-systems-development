	 private double displayDetailsV(ArrayList list) {
        StringBuilder b = new StringBuilder();
        String[] row;
        double total =0;
        b.append("<table>");
        for (Object s : list) {

            row = (String[]) s;
            for (String row1 : row) {
                total += Double.parseDouble(row1);
                b.append(row1);
                
            }
        }
      
        return total;
    }
